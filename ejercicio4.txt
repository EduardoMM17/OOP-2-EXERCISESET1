#include <iostream>
#include <vector>
#include <list>

using std::cout; using std::cin; using std::endl; using std::vector; using std::list;

template<typename container, typename T>
container delete_items(container v1, std::list<T> l1){
    int newCntSize = 0;
    bool isIn;
    for(auto i : v1){
        isIn = false;
        auto it2 = l1.begin();
        while(it2 != l1.end()){
            if(i == *it2)
                isIn = true;
            it2++;
        }
        if(!isIn)
            newCntSize++;
    }
    container v2(newCntSize);
    auto itV2 = v2.begin();
    for(auto i : v1){
        isIn = false;
        auto it2 = l1.begin();
        while(it2 != l1.end()){
            if(i == *it2){
                isIn =true;
            }
            it2++;
        }
        if(!isIn){
            *itV2 = i;
            itV2++;
        }
    }
    return v2;
}

int main() {
    vector<int> v1 = {1,3,4,1,3,2,3,4,6,5};
    auto v2 = delete_items(v1,1);
    for(auto i: v2)
        cout << i << " ";
    //
    cout << endl;
    //
    list<int> v3 ={1,3,4,1,3,2,3,4,6,5};
    list<int> l1 ={1,4};
    auto v4 = delete_items(v3,l1);
    for(auto i : v4)
        cout << i << " ";

    return 0;
}
